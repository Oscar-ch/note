答案：

```mysql
# 1. 显示系统时间

select now();

# 2.查询员工

select
   employee_id,
   last_name,
   salary,
   salary * 1.2 as 'new salary'
from
   employees;

# 3. 排序
select 
   LENGTH(last_name),
   SUBSTR(last_name,1,1) as 首字符
   last_name
from
   employees
order by
   首字符;

# 4. 拼接查询
select
   concat(
      last_name,
      'earns',
      salary,
      'monthly but wants',
      salary * 3) as "Dream Salary"
from
   employees;

# 5. 使用case-when
select
   job_id as job
   case job_id
      when 'AD_PRES' then 'A'
      when 'ST_MAN' then 'B'
      when 'IT_PROE' then 'C'
      when 'SA_REP' then 'D'
      when 'ST_CLERK' then 'E'
      end as frade
from
   employees;
      
```
