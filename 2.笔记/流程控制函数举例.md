# 流程控制函数举例
```mysql
# if
select if(10 < 5, '大'，'小')；

select 
   last_name,
   commission_pct,
   if(commission_pct is null,'没'，'有')
from
   employees; 

# case
/*
case 要判断的字段或表达式
when 常量1 then 要显示的值1或句1
when 常量2 then 要显示的值2或句2
...
else 要显示的值n或语句n;
end
*/ 
# 当switch用
select 
   salary as 原始工资，
   department_id,
   cese department_id
      when 30 then salary * 1.1
      when 40 then salary * 1.2
      when 50 then salary * 1.3
      else salary
      end as 新工资
from employees;

# 当多重if用
select 
   salary,
   case
      when salary > 20000 then 'A'
      when salary > 15000 then 'B'
      when salary > 10000 then 'C'
      else 'D'
      end as工资级别
from employees;
```
